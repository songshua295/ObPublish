# 反射机制

Created: August 8, 2022 9:23 PM Date: August 8, 2022 类别: 反射机制

包类：java.lang.reflect.*

作用：操作字节码文件

  • 相关的重要类
    1. java.lang.Class 代表整个字节码，整个类型
    2. java.lang.reflect.Method 代表字节码中的方法字节码
    3. java.lang.reflect.Constructor 代表字节码中的构造方法字节码
    4. java.lang.reflect.Field 代表字节码中的属性字节码，代表类中的成员变量

# 获取字节码文件的三种方法

  1. 使用静态方法Class.forName()
    1. 参数是包名字符串 ，比如java.lang.String
    2. 赋值给Class对象
    3. 这种方法会导致类中的静态代码块执行
  2. 使用对象.getClass()方法获取
  3. 使用类型.class获取，包括基本类型也可以采用这种方式

import java.util.Date;

public class Test01 {
    public static void main(String[] args) {
        Class c1=null;
        Class c2=null;
//Class.forName()静态方法获取字节码文件
try {
            c1=Class.forName("java.lang.String");//String的字节码文件
c2=Class.forName("java.util.Date");//Date类型的字节码文件
} catch (ClassNotFoundException e) {
            e.printStackTrace();
        }

//java对象的getClass（）方法
String s="abc";
        Class s1=s.getClass();
        System.out.println(s1==c1);//所指向的都是String字节码文件，同一个内存地址，返回true

Date time=new Date();
        Class t1=time.getClass();//获取time对象所指向的class字节码文件 java.util.Date
System.out.println(t1==c2);
        int i=3;

//任何一种数据类型，包括基本数据类型，都有.class属性
Class c3=int.class;
        Class c4=Date.class;
        System.out.println(c4==c2);
        System.out.println(c4==t1);
    }
}

import java.lang.reflect.Field;
import java.lang.reflect.Modifier;

public class Test06 {
    public static void main(String[] args) throws Exception {
        //    获取整个类
        Class student = Class.forName("com.reflect.test.Student");
        System.out.print("\n完整类名:" + student.getName());
        System.out.print("\n简单类名:" + student.getSimpleName());

        //    获取类中public修饰的字段Fields
        Field[] fields = student.getFields();
        System.out.print("\n\npublic修饰的字段：" + fields.length);
        System.out.print("\n各个public修饰字段名称是：==================");
        for (Field f : fields) {
            System.out.println(f.getName());
        }

        //    获取所有声明的字段Fields
        Field[] fields1 = student.getDeclaredFields();
        System.out.print("\n\n已经声明的字段个数：" + fields1.length);
        System.out.print("\n已经声明的字段===============");
        for (Field f : fields1) {
            System.out.println(f.getName());
        }

// 更多的方法
        System.out.println("常用方法的使用==========");
        for (Field f : fields1) {
            // 字段修饰符
            //    获取修饰符标识数字
            int modifyInt = f.getModifiers();
            //    转换修饰符数字为本来的字符串，比如public private等等
            String modifyString = Modifier.toString(modifyInt);
            System.out.print(modifyString);
            System.out.print(" ");

            //    字段返回类型,也就是对象的属性，比如int，Student类
            Class fieldType = f.getType();
            System.out.print(fieldType.getSimpleName());//获取简单类名
            System.out.print(" ");
            //    获取字段名称
            System.out.print(f.getName());
            System.out.print(" ");
            System.out.println();
        }

    }
}

import com.reflect.test.Student;

import java.lang.reflect.Field;
import java.lang.reflect.Modifier;

public class Test07 {
    public static void main(String[] args) throws Exception {
        // 用来保存代码的内容
        StringBuilder s = new StringBuilder();

        //    获取类字节码文件
        Class studentClass = Class.forName("com.reflect.test.Student");

        //    首行的类声明
        s.append(Modifier.toString(studentClass.getModifiers()) + " class " + studentClass.getSimpleName() + "{\n");
        // 获取各个字段Fields
        Field[] fields = studentClass.getDeclaredFields();
        for (Field f : fields
        ) {
            s.append("\t");
            //                           修饰符                 变量类型           变量名
            s.append(Modifier.toString(f.getModifiers()) + " " +f.getType().getSimpleName()+" "+f.getName()+";\n");
        }
        s.append("}");
        System.out.println(s);
    }
}

import com.reflect.test.Student;

import java.lang.reflect.Field;

public class Test09 {
    public static void main(String[] args)throws Exception {
//不使用反射机制
Student s1=new Student();
        s1.no=12;//设置属性
System.out.println(s1.no);//访问属性
//使用反射机制,会增加复杂性，但是能够增加灵活性
//获取类
Class c2=Class.forName("com.reflect.test.Student");
//创建无参构造方法
Object s3=c2.newInstance();
//获取no属性字段
Field noField=c2.getDeclaredField("no");
//给s3对象no字段赋值
noField.set(s3,45);
//获取s3对象nno字段的值
System.out.println(noField.get(s3));

//访问私有属性，无法直接访问，需要打破封装，但是打破封装之后可能会出现安全问题
Field nameField=c2.getDeclaredField("name");
//打破封装
nameField.setAccessible(true);
        nameField.set(s3,"蔡徐坤");//赋值
System.out.println(nameField.get(s3));

    }
}

import java.lang.reflect.Field;
import java.lang.reflect.Method;
import java.lang.reflect.Modifier;
import java.lang.reflect.Parameter;

public class Test11 {
    public static void main(String[] args) throws Exception {
        StringBuilder s=new StringBuilder();
/*类声明 */
Class aClass=Class.forName("com.reflect.test.Person");
        String p1= Modifier.toString(aClass.getModifiers());//修饰符
String c1=aClass.getSimpleName();//获取简单类名
s.append(p1+" class "+c1+"{\n");
/*获取字段 */
Field[] fields=aClass.getDeclaredFields();
        for (Field f:fields
             ) {
//获取字段修饰符
s.append("\t");
            s.append(Modifier.toString(f.getModifiers()));
            s.append(" ");
//获取字段类型
s.append(f.getType().getSimpleName());
            s.append(" ");
//获取字段名称
s.append(f.getName());
            s.append(";\n");
        }

/*获取方法 */
Method[] methods=aClass.getDeclaredMethods();
        for (Method m:methods
             ) {
//获取方法修饰符
s.append("\t");
            s.append(Modifier.toString(m.getModifiers()));
            s.append(" ");
//获取方法返回值类型
s.append(m.getReturnType());
            s.append(" ");
//获取方法名称
s.append(m.getName());
            s.append("(");
//获取形参列表
Class[] parameters=m.getParameterTypes();
            for (Class c:parameters
                 ) {
                s.append(c.getSimpleName());
                s.append(",");
            }
            s.deleteCharAt(s.length()-1);//删除因为循环多写的那个逗号
s.append("){}\n");//方法体结尾
}

/*类声明结尾 */
s.append("}");

        System.out.println(s);
    }
}

  3. 调用方法对象的方法，method.invoke(对象名，方法形参…)
  4. 可以设置一个对象用于接受返回值 Object o

import com.reflect.test.Person;

import java.lang.reflect.Method;

public class Test12 {
    public static void main(String[] args) throws Exception{
/*正常的调用方法 */
    //创建对象
Person p1=new Person();
//调用方法
System.out.println(p1.login("admin","123")?"登录成功":"登录失败");
        p1.logout();

/*采用反射机制调用方法 */
    //    1.获取类
Class aClass=Class.forName("com.reflect.test.Person");
//    2.创建对象
Object o=aClass.newInstance();//创建无参对象
//    3.查找方法
Method method=aClass.getDeclaredMethod("login", String.class, String.class);//方法名，形参列表所使用的类
//    4.调用方法
Object o2=method.invoke(o,"admin","123");
        System.out.println(o2);

//调用另一个方法
Method method1=aClass.getDeclaredMethod("logout");
        method1.invoke(o);
    }
}

package com.reflect.test;

public class Vip {
    int no;
    String name;
    String birth;
    boolean sex;

    public Vip() {
    }

    public Vip(int no) {
        this.no = no;
    }

    public Vip(int no, String name) {
        this.no = no;
        this.name = name;
    }

    public Vip(int no, String name, String birth) {
        this.no = no;
        this.name = name;
        this.birth = birth;
    }

    public Vip(int no, String name, String birth, boolean sex) {
        this.no = no;
        this.name = name;
        this.birth = birth;
        this.sex = sex;
    }

    @Override
    public String toString() {
        return "Vip{" +
                "no=" + no +
                ", name='" + name + '\'' +
                ", birth='" + birth + '\'' +
                ", sex=" + sex +
                '}';
    }
}
