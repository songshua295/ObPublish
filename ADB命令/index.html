# 普通卸载

1
2
3
4
5
6

//所有软件的包名
adb shell pm list packages

//卸载命令
adb uninstall 包名  #或者
adb shell pm uninstall --user 0 包名

# 删除的系统软件的另一种方式

1
2
3
4
5

adb root。   #在终端输入该命令获得root权限。
adb remount。  #重新挂载，获得对读写权限。
adb shell。   #进入adb命令环境。
cd /ystem/app #system/app目录，一般系统apk都安装在这个目录。
rm -rf Browser.apk   #卸载那个文件，没有尝试过

# 另一种离谱的方式

并不会删除软件，但是会导致安装上的软件所有数据全部删除，0b的无数据软件 使其无法正常工作，无法工作

1
2
3
4

adb shell #进入shell命令
cd /system/     #这里记不得目录了，好像是sdcard目录
cd Android/data/
rm -rf 包名

//获取包名地址
C:\\Users\\ShenWeiXing>adb shell pm path com.xxx.xxx
package:/data/app/包名/base.apk

//挂载系统读写权限
C:\\Users\\ShenWeiXing>adb remount
remount succeeded

//删除包
C:\\Users\\ShenWeiXing>adb shell rm /data/app/com.xxx.xxx-1/base.apk

//重启后ok
C:\\Users\\ShenWeiXing>adb reboot
