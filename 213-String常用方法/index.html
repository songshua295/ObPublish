public class Test03 {  
    public static void main(String[] args) {  
        //charAt 返回字符串指定下标的字符  
        String a="我是一个中国人";  
        System.out.println(a.charAt(2));  
        System.out.println("===============");  
  
        //compareTo 按照字典进行比较,按位比较,0就相等,正数或者负数表示前面和后面第一个不相同的数字的大小差值,能比出来,后面的就不比较了  
        System.out.println("abc".compareTo("abd"));  
        System.out.println("abc".compareTo("abb"));  
        System.out.println("abc".compareTo("cba"));  
        System.out.println("abc".compareTo("abc"));  
        System.out.println("abc".compareTo("efz"));  
        System.out.println("abc".compareTo("a"));  
        System.out.println("=================");  
  
        //contains(CharSequence s),前面的字符串是否包含后面的字符串  
        System.out.println("我是你爸爸的儿子的大帅比".contains("儿子"));  
        System.out.println("==============");  
  
        System.out.println("endsWith(String suffix) 此字符串是否以指定的后缀结尾。");  
        System.out.println("我是大帅比".endsWith("大帅比"));  
        System.out.println("我是蔡徐坤".endsWith("还好吧"));  
        System.out.println("startsWith(String prefix) 此字符串是否以指定的前缀开始");  
        System.out.println("javadasklfjkalsfjakok".startsWith("java"));  
        System.out.println("==================");  
  
        //equals() 和      equalsIgnoreCase(String str)        System.out.println("abc".equals("abc"));  
        System.out.println("abc".equals("abC"));  
        System.out.println("abc".  equalsIgnoreCase("abC"));//忽略大小写的比较  
  
    //    getBytes()  把字符串对象转换成byte数组  
        byte[] byte1="abc".getBytes();  
        System.out.println(byte1[2]);  
  
        // indexOf(String str) 返回指定子字符串第一次出现的字符串内的索引。  
        //lastIndexOf(String str)  最后一次出现的索引  
        System.out.println("凉风不顾归尘仆仆的行者零落了世间所有的风华，我在天涯征程之外寻寻觅觅那一树的花开，冷香随着秋风袭卷了草木荣枯的来去归途。".indexOf("世间所有的风华"));  
        System.out.println("你是谁,我是谁".lastIndexOf("是谁"));  
        // isEmpty() length是否为0,不包括null  
        System.out.println("isEmpty=======");  
        System.out.println("ab".isEmpty());  
        System.out.println("".isEmpty());  
        String string1=new String();  
        //string1=null;  //下面会出现空值异常,底层调用的length方法  
        System.out.println(string1.isEmpty());  
  
  
        //判断字符串长度是length()方法,判断数组长度是length属性  
        System.out.println("hello".length());  
  
        // replace(char oldChar, char newChar) 替换字符串  
        System.out.println("我是大帅比,你是不是?".replace("是","像"));  
  
        //split(String regex)  用字符串分割为数组  
        String[] sp1="帅哥,小姐,美女,野兽,帅比,儿子,孙子,玫瑰,显化".split(",");  
        for (int i = 0; i <sp1.length ; i++) {  
            System.out.print(sp1[i]+" ");  
        }  
  
        System.out.println("substring(int beginIndex) 返回该字符串指定下标开始的子字符串");  
        System.out.println("https://www.baidu.com".substring(5));//下标为5开始的后面字符串,包括5  
        System.out.println("https://www.baidu.com".substring(4,6));//下标为4开始,下标为6-1结束,左闭右开  
  
        System.out.println("toCharArray() 将此字符串转换为新的字符数组。");  
        char[] chars2="我是大帅哥".toCharArray();  
        for (int i = 0; i < chars2.length; i++) {  
            System.out.println(chars2[i]);  
        }  
  
        System.out.println("toLowerCase() 转换成小写");  
        System.out.println("ABdfdsfDDSFS".toLowerCase());  
        System.out.println("\ttoUpperCase() 转换成大写");  
        System.out.println("abcdefgHJ".toUpperCase());  
  
  
        System.out.println("\ttrim()\n" +  
                "返回一个字符串，其值为此字符串，并删除任何前导和尾随空格。");  
        System.out.println("   sjkfjslf  fskjfs  fsjfks".trim());  
  
        System.out.println("valueOf(boolean b) 把不是字符串的转换成字符串");  
        String s7=String.valueOf(true);  
        System.out.println(s7);//是字符串，不是boolean类型了  
        System.out.println(String.valueOf(new Customer()));//当是对象的时候，会调用toString方法  
    }  
  
}

    public String toString() {  
        return "我是一个vip客户";  
    }  
}
