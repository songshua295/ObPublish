import java.sql.*;  
  
public class JDBCTest05 {  
    public static void main(String[] args) {  
    //    变量预设  
        Connection connection=null;//连接的对象  
        Statement statement=null;//发送sql语句的对象  
        ResultSet resultSet=null;//结果集  
  
        try {  
        //    注册驱动  
            Class.forName("com.mysql.jdbc.Driver");  
        //    建立连接  
            connection= DriverManager.getConnection("jdbc:mysql://localhost:3306/bjpowernode","root","root");  
        //    获取发送sql语句的对象  
            statement=connection.createStatement();  
        //    发送sql语句  
            resultSet=statement.executeQuery("select * from dept");//专门执行DQL的方法,返回值是ResultSet  
        //    处理查询集合  
            while (resultSet.next()){  
                //getString 可以将数据库中的内容以String的方式获取  
                //可以使用列数，从1开始，不是从0就开始  
                int deptno=resultSet.getInt("deptno");  
                String dname=resultSet.getString("dname");  
                String loc=resultSet.getString("loc");  
                System.out.println(deptno+" "+dname+" "+loc);  
            }  
  
        } catch (ClassNotFoundException e) {  
            e.printStackTrace();  
        } catch (SQLException e) {  
            e.printStackTrace();  
        }finally {  
        //    6.释放资源  
            if (resultSet!=null){  
                try {  
                    resultSet.close();  
                } catch (SQLException e) {  
                    e.printStackTrace();  
                }  
            }  
            if (statement!=null){  
                try {  
                    statement.close();  
                } catch (SQLException e) {  
                    e.printStackTrace();  
                }  
            }  
            if (connection!=null){  
                try {  
                    connection.close();  
                } catch (SQLException e) {  
                    e.printStackTrace();  
                }  
            }  
        }  
    }  
}
