     * 初始化数据  
     *  
     * @param path 路径  
     * @param list 列表  
     */  
    public static void initData(String path, List<?> list) {  
        // 路径提取  
        File directory = new File(path.split("/")[0]);  
        File file = new File(path);// 初始化文件存放路径  
        ObjectOutputStream oos=null;//对象输出流  
        //   判断文件夹是否存在  
        if (!directory.exists()) {  
            directory.mkdir();  
        }  
        //    判断文件是否存在  
        if (!file.exists()){  
            try {  
                file.createNewFile();  
                oos=new ObjectOutputStream(new FileOutputStream(file));  
                oos.writeObject(list);//写入对象数据  
            } catch (IOException e) {  
                e.printStackTrace();  
            }finally {  
                try {  
                    oos.close();  
                } catch (IOException e) {  
                    e.printStackTrace();  
                }  
            }  
        }  
  
    }  
}

package com.bjpowernode.dao.Impl;  
  
import com.bjpowernode.bean.PathConstant;  
import com.bjpowernode.bean.User;  
import com.bjpowernode.dao.UserDao;  
  
import java.io.*;  
import java.util.ArrayList;  
import java.util.List;  
  
public class UserDaoImpl implements UserDao {  
    /**  
     * 选择  
     *  
     * @return {@link List}<{@link User}>  
     */    @Override  
    public List<User> select() {  
        try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(PathConstant.User_Path))) {  
            //    读取  
            List<User> list = (List<User>) ois.readObject();  
            return list;  
        } catch (Exception e) {  
            // throw new RuntimeException(e);  
            e.printStackTrace();  
        }  
        // 如果异常，就返回一个无用户对象的空列表  
        return new ArrayList<>();  
    }  
  
    /**  
     * 添加  
     *  
     * @param user 用户  
     */  
    @Override  
    public void add(User user) {  
        ObjectInputStream ois = null;  
        ObjectOutputStream oos = null;  
        List<User> list = null;  
        try {  
            //    读取硬盘中的用户信息  
            ois = new ObjectInputStream(new FileInputStream(PathConstant.User_Path));  
  
            list = (List<User>) ois.readObject();// 读取用户信息  
            // 判断list是否有  
            if (list != null) {  
                // 设置新添加的用户id  
                User lastUser = list.get(list.size() - 1);  
                user.setId(lastUser.getId() + 1);  
                //    添加到list中  
                list.add(user);  
            } else {  
                //    当list为空的时候，就需要一个空的去装  
                list = new ArrayList<>();  
                user.setId(0001);  
                list.add(user);  
  
            }  
            //    持久化到本地硬盘  
            oos = new ObjectOutputStream(new FileOutputStream(PathConstant.User_Path));  
  
            oos.writeObject(list);  
            oos.flush();  
  
        } catch (Exception e) {  
            throw new RuntimeException("添加用户失败");  
        } finally {  
            // 释放资源  
            try {  
                if (ois != null) {  
                    ois.close();  
                }  
                if (oos != null) {  
                    oos.close();  
                }  
            } catch (IOException e) {  
                e.printStackTrace();  
            }  
        }  
    }  
}

/**  
 * 复制对象属性  
 *  
 * @param origin 源对象  
 * @param dest   需要复制的对象属性  
 */  
public static void populate(Object origin, Object dest) {  
    try {  
        /* 通过反射机制复制对象的属性 */        if (origin.getClass()!=dest.getClass()){  
            throw  new RuntimeException("必须是一样的类型才可以进行对象属性复制！！！");  
        }  
  
        Class<?> clazz=origin.getClass();//获取需要修改值的类文件  
        Field[] fields=clazz.getDeclaredFields();//获取属性字段  
        for (Field f:fields  
        ) {  
            //不对序列号id进行复制  
            if ("serialVersionUID".equals(f.getName())){  
                continue;  
            }  
            //打破封装  
            f.setAccessible(true);  
            //将dest的值复制给f  
            f.set(origin,f.get(dest));  
        }  
    }catch (IllegalAccessException e) {  
        throw new RuntimeException(e);  
    }  
}

            throw new RuntimeException(e);

                conditionList = conditionList.stream().filter(b -> b.getBookName().equals(book.getBookName())).collect(Collectors.toList());  
  
            }  
            return conditionList;  
        }  
    } catch (Exception e) {  
        e.printStackTrace();  
    } finally {  
        try {  
            if (ois != null) {  
                ois.close();  
            }  
  
        } catch (IOException e) {  
            e.printStackTrace();  
        }  
    }  
    // 出现了异常，就返回一个空的  
    return list;  
}

@Override  
public void add(Book Book) {  
    ObjectInputStream ois = null;  
    ObjectOutputStream oos = null;  
    List<Book> list = null;  
    try {  
        //    读取硬盘中的书籍信息  
        ois = new ObjectInputStream(new FileInputStream(PathConstant.Book_Path));  
  
        list = (List<Book>) ois.readObject();// 读取书籍信息  
        // 判断list是否有  
        if (list != null) {  
            // 设置新添加的书籍id  
            Book lastBook = list.get(list.size() - 1);  
            Book.setId(lastBook.getId() + 1);  
            //    添加到list中  
            list.add(Book);  
        } else {  
            //    当list为空的时候，就需要一个空的去装  
            list = new ArrayList<>();  
            Book.setId(0001);  
            list.add(Book);  
  
        }  
        //    持久化到本地硬盘  
        oos = new ObjectOutputStream(new FileOutputStream(PathConstant.Book_Path));  
  
        oos.writeObject(list);  
        oos.flush();  
  
    } catch (Exception e) {  
        throw new RuntimeException("添加书籍失败");  
    } finally {  
        // 释放资源  
        try {  
            if (ois != null) {  
                ois.close();  
            }  
            if (oos != null) {  
                oos.close();  
            }  
        } catch (IOException e) {  
            e.printStackTrace();  
        }  
    }  
}

/**  
 * 图书类型数量统计  
 *  
 * @return {@link Map}<{@link String}, {@link Integer}>  
 */@Override  
public Map<String, Integer> bookTypeCount() {  
    try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(PathConstant.Book_Path))) {  
        //    获取所有的图书信息  
        List<Book> list = (List<Book>) ois.readObject();  
    //    对图书根据书籍类型进行分组  
        Map<String,List<Book>> collect=list.stream().collect(Collectors.groupingBy(Book::getType));  
    //    遍历分别进行统计  
        HashMap<String ,Integer> map=new HashMap<>();  
        Iterator<Map.Entry<String,List<Book>>> iterator=collect.entrySet().iterator();//迭代器  
        while (iterator.hasNext()){  
        //    将统计结果放入到map中  
            Map.Entry<String,List<Book>> next=iterator.next();//具体的书籍分类信息  
            map.put(next.getKey(),next.getValue()==null?0:next.getValue().size());//写入统计书籍类型的数量  
        }  
        //返回结果  
        return map;  
    } catch (Exception e) {  
        e.printStackTrace();  
        throw new RuntimeException("统计图书异常~");  
    }  
}

/**  
 * 查询  
 *  
 * @param lend 书  
 * @return {@link List}<{@link Lend}>  
 */@Override  
public List<Lend> select(Lend lend) {  
    ObjectInputStream ois = null;  
    List<Lend> list = new ArrayList<>();  
    try {  
        //    读取借阅信息  
        ois = new ObjectInputStream(new FileInputStream(PathConstant.Lend_Path));  
        list = (List<Lend>) ois.readObject();  
        //当lend为空的时候，就返回所有的借阅信息  
        if (lend==null||"".equals(lend.getBook().getBookName()) && "".equals(lend.getBook().getIsbn())){  
            return list;  
        }else{  
            //处理结果  
            List<Lend> conditionList=new ArrayList<>();  
            //通过书籍名称获取借阅信息  
            if(!"".equals(lend.getBook().getBookName())){  
                conditionList=list.stream().filter(l->l.getBook().getBookName().equals(lend.getBook().getBookName())).collect(Collectors.toList());  
            }  
            // 通过书籍ISBN号获取借阅信息  
            if(!"".equals(lend.getBook().getIsbn())){  
                conditionList=list.stream().filter(l->l.getBook().getIsbn().equals(lend.getBook().getIsbn())).collect(Collectors.toList());  
            }  
            //联合查询  
            if (!"".equals(lend.getBook().getIsbn())&&!"".equals(lend.getBook().getBookName())){  
                conditionList=conditionList.stream().filter(l->l.getBook().getBookName().equals(lend.getBook().getBookName())).collect(Collectors.toList());  
            }  
            //返回借阅结果集  
            return  conditionList;  
        }  
  
  
    } catch (Exception e) {  
        e.printStackTrace();  
    } finally {  
        try {  
            if (ois != null) {  
                ois.close();  
            }  
  
        } catch (IOException e) {  
            e.printStackTrace();  
        }  
    }  
    // 出现了异常，就返回一个空的  
    return list;  
}
