    # 相关

    1. 类继承：猫狗与宠物之间的继承关系
    2. 向下继承：向下继承，如果需要调用子类独有的方法（pig的睡觉方法，在pet中没有），需要强制转换类型，前提是两个类继承关系，且向下转型才必须强制转型
    3. 独有实例方法的调用
    4. 向上继承的创建对象：Pet p2=new Pig();
    5. 多态：宠物的多个具体事物，有不同的吃的方法，主人通过调用喂养方法，去访问宠物的吃
    6. 方法覆盖override：由于子类有不同的eat方法，在实际调用的时候，会访问其具体对象的eat方法

    package a007方法覆盖和多态.多态例子动物喂养;


public class Test {
    public static void main(String[] args) {
        System.out.println("主人喂养宠物实例：");
    
        Master cxk=new Master("蔡徐坤");//创建主人
        System.out.println(cxk.getName());
        Pet p1=new Pet();//创建宠物
        System.out.println("\n"+cxk.getName()+"开始喂宠物：");
        cxk.feed(p1);

        Dog d1=new Dog("旺财");//创建狗对象
        System.out.println("\n"+cxk.getName()+"开始喂狗："+d1.getName());
        cxk.feed(d1);

        Cat c1 = new Cat("汤姆");//创建猫对象
        System.out.println("\n"+cxk.getName()+"开始喂猫："+c1.getName());
        cxk.feed(c1);

        System.out.println("Pig对象");
        Pet p2 = new Pig("佩奇");
        System.out.print(p2.getName());
        Pig p3=(Pig)p2;
        p3.sleep();

        Pig p4=new Pig("乔治");
        System.out.print(p4.getName());
        p4.sleep();
                

    }
}


// Master class
class Master{
    private String name;
    //get and set
    public String getName() {
        return name;
    }
    public void setName(String name) {
        this.name = name;
    }
    //constructor
    public Master(){}
    public Master(String name) {
        this.name = name;
    }


    // 喂养宠物
    public void feed(Pet pet){
        pet.eat();
    }
}

// class Pet
class Pet{
    private String name;
    
    // constructor
    public Pet(){}
    public Pet(String name) {
        this.name = name;
    }


    public String getName() {
        return name;
    }


    public void setName(String name) {
        this.name = name;
    }


    // 宠物吃东西
    public void eat(){
        System.out.println("宠物吃东西");
    }
}

// Dog class
class Dog extends Pet{
    //re constructor

    public Dog(String name) {
        super(name);
    }

    public Dog() {
    }

    // 狗儿啃骨头
    public void eat(){
        System.out.println("狗儿啃骨头");
    }
}

// Cat class
class Cat extends Pet{
    
    public Cat() {
    }

    public Cat(String name) {
        super(name);
    }

    public void eat(){
        System.out.println("猫儿吃鱼");
    }
}

class Pig extends Pet{
    public Pig(){
    }
    public Pig(String name){
        super(name);
    }
    // 猪的独有方法
    public void sleep(){
        System.out.println("猪在睡觉");
    }
}
