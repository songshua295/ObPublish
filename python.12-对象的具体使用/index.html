    # 封装

    是为了数据安全，避免外部随意访问 封装方法：在创建构造函数（__init__)的时候，使用self.__属性名对其进行封装 封装之后，可以创建实例方法来对其属性进行操作，也可以使用_类名__属性名来对其进行强行访问

    class Student:
    def __init__(self,name,age):
        self.name=name
        self.__age=age #不希望被外部访问，就封装该属性
    # 使用实例访问去获取属性
    def getage(self):
        return self.__age
s1=Student("张三",23)
print(s1.name)
# print(s1.__age) #无法直接访问
print(s1.getage()) #通过实例访问去访问

#如果没有实例访问，硬要去访问
print(dir(s1)) #获取s1实例对象的所有属性
print(s1._Student__age) #使用_类名__属性，还是能够访问

    class Person:

    class Animal:

        print("动物吃食物")
    def __add__(self,other):
        return self.name+other.name

        print("狗吃骨头")

    def __init__(self):
        self.name="default"
    def eat(self):
        print("猫吃鱼")
    def _len_(self):
        return 1

    class Person(object):
    def __new__(cls,*args,**kwargs):
        print(f'__new__被调用，cls的id为{id(cls)}')
        obj=super().__new__(cls)
        print(f'__new__创建的对象object的id为{id(obj)}')
        return obj
    def __init__(self,name,age):
        print(f'__init__被调用，id为{id(self)}')
        self.name=name
        self.age=age

print(f'object这个类对象的id为{id(object)}')
print(f'Person这个类对象的id为{id(Person)}')
#实例对象

    # 浅拷贝与深拷贝

    # 浅拷贝

    Python拷贝一般都是浅拷贝，拷贝时，对象包含的子对象内容不拷贝，因此，源对象与拷贝对象会引用同一个子对象。（拷贝的对象中子对象的引用）

    # 深拷贝

    deepcopy方法，在拷贝对象时候，对象包含的子对象也会被拷贝。

    • 两个的差别都是所拷贝的对象中存在的子对象是否拷贝，浅拷贝的子对象是公用同一个的，而深拷贝的子对象也会被拷贝一份。

      class Computer:
    def __init__(self,cpu,disk):
        self.cpu=cpu
        self.disk=disk
class Disk:
    pass
class CPU:
    pass
#创建一个compouter
disk1=Disk()
cpu1=CPU()
computer1=Computer(disk1,cpu1)
print("disk1",disk1)
print("cpu1",cpu1)
print("compouter1",computer1)
#浅拷贝
import copy
computer2=copy.copy(computer1)
print("computer2",computer2)
print(computer2.cpu)
print(computer2.disk)
print(computer1.disk is computer2.disk) #子对象没有拷贝，所以还是同一个子对象
#深拷贝
computer3=copy.deepcopy(computer1)
print(computer3 is computer1,id(computer3),id(computer1))
print(computer1.cpu is computer3) #子对象也被拷贝，所以内存地址和id也不同
